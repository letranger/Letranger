5 動機前後測 t 檢定

 3: 
 4: library(readr)
 5: library(plyr)
 6: data <- read_csv("motivation2.csv")
 7: 
 8: cols_to_test <- c("Q1","Q18","Q24","Q26","Q8","Q12","Q14","Q34","Q5","Q11","Q19","Q25","Q28","Q29","Q2","Q4","Q10","Q15","Q20","Q27","Q30","Q33","Q7","Q13","Q17","Q22","Q32","Q6","Q23","Q35","Q3","Q9","Q16","Q21","Q31"
 9: )
10: 
11: results <- ldply(
12:   cols_to_test,
13:   function(colname) {
14:     t_val = t.test(data[[colname]] ~ data$EXEC)$statistic
15:     p_val = t.test(data[[colname]] ~ data$EXEC)$p.value
16:     return(data.frame(題號=colname, t_value=t_val, p_value=p_val))
17:   }
18: )
19: results
